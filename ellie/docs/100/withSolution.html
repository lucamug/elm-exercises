g o al

(\f1 -> f1 "g") (\s1 f2 -> f2 (s1 ++ "o")) (\s2 -> s2 ++ "al")   -- expanded to definitions

((\f1 -> f1 "g") (\s1 f2 -> f2 (s1 ++ "o")) (\s2 -> s2 ++ "al")) -- added parens, it goes (g o) al, not g (o al)

((\s1 f2 -> f2 (s1 ++ "o") "g") (\s2 -> s2 ++ "al")              -- applied middle fn to the first fn
